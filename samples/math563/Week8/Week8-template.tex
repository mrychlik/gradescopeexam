  \begin{question}
    Let $T_1,T_2,\ldots T_n$ be non-negative, independent continuous random variables and let
    $$V = \min(T_1,T_2,\ldots,T_n)$$ be their minimum. Let $f_j(t)=f_{T_j}(t)$ be their pdf
    and $F_j(t)$ be the corresponding cdf. Let $G_j=1-F_j$ be the tail probability.
    Let $I$ be the index of the smallest of $T_i$, i.e.
    $I = i$ iff $\forall j\neq i \,:\, T_i < T_j$
    {\color{magenta}All answers must be expressed in terms of $f_j$ and $G_j$.}
    \begin{parts}
      \part Find the tail probability $\P(V>t)$.
      \begin{solutionorbox}[2.5in]
      \end{solutionorbox}
      \part Find the probability function of $I$, i.e., $\P(I=i)$.
      \begin{solutionorbox}[2.5in]
      \end{solutionorbox}
      \part Find $ \P(V > t | I = i) $. Explain your reasoning, paying attention to proper notation.
      \begin{solutionorbox}[8in]
      \end{solutionorbox}
      \part Use the Bayes formula for conditional expectations to find $ \P(I=i | V = t)$.
      Explain your reasoning, paying attention to proper notation.
      \begin{solutionorbox}[7.8in]
      \end{solutionorbox}
    \end{parts}
  \end{question}
